int check_anagram(string str_1, string str_2) {
    
    int i, j, len;
    char ch;
    char buf[256] = {0};
    
    // std::fill_n(array, 100, -1);
    
    i = len = 0;
    
    while(str_1[i]){
        ch = str_1[i];
        buf[ch] += 1;    
        len++;
        i++;   
    }
    
    j = 0;
    
    while(str_2[j] != '\0' && len > 0) {
        ch = str_2[j];        
        if(buf[ch] != 0) {
            buf[ch] -= 1;
            len--;
            j++;
        }
        else {
            break;
        }    
    }    
    
    if(len > 0 && str_2[j] != '\0') {
        return -1;
    }
    return 0;
}

	int ret = check_anagram(str_1, str_2);
	
	if(ret == 0)

	    cout <<"Not Anagram";
	else {

	}

}
