// A 32 bit unsigned integer square fit in 16 bit integer
int sqr_root(int N) 
{
    unsigned int low = 0;
    unsigned int end = 65536;
    
    while(low+1 < end) {
        unsigned int mid = low + (end - low)/2;
        
        if(mid*mid == N) {
            return mid;
        }
        else if(mid*mid > N) {
            end = mid;
        }
        else {
            low = mid;
        }
    }
    return low;
}

	int ret = sqr_root(N);
	
	if(ret == -1) {
	    cout <<"Key not found";
	}
	else {
	    cout <<"square root is " << ret;
	}
	return 0;
}
