int binary_search(int *p, int size, int srch)
{
	int low, hi, mid;
	low = 0;
	hi = size - 1;
	
	while(low <= hi)
	{
		 mid = low + (hi - low)/2;
		 if(p[mid] == srch)
		 {
		 	return mid;
		 }
		 else if(srch < p[mid]) // Discard all the Elem right to the mid 
		 {
		 	hi = mid - 1;
		 }
		 else                // // Discard all the Elem left to the mid 
		 {
		 	low = mid + 1;
		 }
	}
   return 1;
}

int main(void) {
	// your code goes here
	int ar_size, ret;
	
	scanf("%d",&ar_size);
	
	int ar[ar_size], ar_i, srch;
	
	for(ar_i = 0; ar_i < ar_size; ar_i++)
	{
		scanf("%d",&ar[ar_i]);
	}
	scanf("%d",&srch);
    ret = binary_search(ar, ar_size, srch);
    if(ar[ret] == srch)
    {
    	printf("\n Elem found at mid value = %d ",ret);
    }
    else
    {
    	printf("\n srch is not found in the sorted array");
    }
    return 0;
}
