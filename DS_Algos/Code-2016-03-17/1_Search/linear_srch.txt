#include <stdio.h>
#include<stdlib.h>
#include<string.h>
#include<assert.h>

void linear_search(int *p, int size)
{
	int i, flag, srch;
	
	scanf("%d",&srch);
	
	for(i = 0; i < size; i++){
		 if(p[i] == srch){
		 	flag = 1;
		 	break;
		 }
	}
	
    if(flag == 1) 
      printf("\n %d no Elem found ", srch);
    else
      printf("\n %d no Elem not found ", srch);
}

int main(void) {
	// your code goes here
	int ar_size;
	
	scanf("%d",&ar_size);
	
	int ar[ar_size], ar_i;
	
	for(ar_i = 0; ar_i < ar_size; ar_i++){
		scanf("%d",&ar[ar_i]);
	}
    linear_search(ar, ar_size);
    
	return 0;
}
