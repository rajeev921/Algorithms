char* dup_str_cpy(char *str)
{
    char *dup,*temp;
    int size, len = 0;
    
    if(str == NULL)
        return NULL;
    else if(str[0] == ' ')
        return NULL;
    else {
        size = strlen(str);
        temp = str;
        dup = (char *)malloc(sizeof(char)*size);
        assert(dup != NULL);
        
        while(*temp){
            dup[len] = *temp;
            len++;
            temp++;
        }
        dup[len] = '\0';
    }
    puts(dup);
    return dup;
}
