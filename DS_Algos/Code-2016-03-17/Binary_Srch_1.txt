using namespace std;

int binary_search(int a[], int N, int key) 
{
    int low = 0;
    int end = N-1;
    
    while(low <= end) {
        int mid = low + (end - low)/2;
        
        if(a[mid] == key) {
            return mid;
        }
        else if(key > a[mid]) {
            low = mid+1;
        }
        else {
            end = mid-1;
        }
    }
    return -1;
}

int main() {
	// your code goes here
	int a[] = {1, 3, 4, 8, 10, 12, 14, 16, 20, 25, 28, 30, 35, 39, 40, 50};
	int N = sizeof(a)/sizeof(a[0]);
	int key;
	
	cin >> key;
	
	int ret = binary_search(a, N, key);
	
	if(ret == -1) {
	    cout <<"Key not found";
	}
	else {
	    cout <<"key found at pos " << ret;
	}
	return 0;
}
